' このファイルの拡張子をvbsにし、クリックすると使えます。
' (5の倍数-1)分の時に、「周囲の状況を確認しましたか?」と通知されます。

MsgBox"周囲確認タイマーを起動しました。",,"周囲確認タイマー"
Do
	WScript.Sleep 300000-(1000*Timer+60000)Mod 300000
	While MsgBox("周囲の状況を確認しましたか?",308,"周囲の状況を確認しましょう")=7
		MsgBox"周囲の状況を確認しましょう。",16,"周囲の状況を確認しましょう"
	Wend
Loop While MsgBox("作業を続けますか?",36,"周囲確認タイマー")=6
